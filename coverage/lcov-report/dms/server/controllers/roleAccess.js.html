const rbac = require('easy-rbac').create(
  {
    user: { // Role name
      can: ['doc:create', { // list of allowed operations
        name: 'doc:update',
        when: function (params, callback) {
          setImmediate(callback, null, params.Id === params.ownerId);
        } },
      ],
    },
    admin: {
      can: ['doc:update'],
    },
  }
);
